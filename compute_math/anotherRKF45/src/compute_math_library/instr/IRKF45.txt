      SUBROUTINE ORBIT(T,Y,YP)
      REAL T,Y(4),YP(4),R,ALFASQ
      COMMON ALFASQ
      R=Y(1)*Y(1)+Y(2)*Y(2)
      R=R*SQRT(R)/ALFASQ
      YP(1)=Y(3)
      YP(2)=Y(4)
      YP(3)=-Y(1)/R
      YP(4)=-Y(2)/R
      RETURN
      END
